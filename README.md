    <script>
        const answers = { e1:"B", e2:"B", g1:"A", g2:"B", i1:"A", p1:"A", m1:"A", m2:"A" };

        function getResult() {
            let score = 0;
            const total = Object.keys(answers).length;
            const form = new FormData(document.getElementById('megaExam'));
            for(let k in answers) { if(form.get(k) === answers[k]) score++; }
            const p = Math.round((score/total)*100);
            document.getElementById('finalMarks').innerText = p + "%";
            document.getElementById('statusTxt').innerText = p >= 50 ? "Qualified" : "Not Qualified";
            document.getElementById('resPortal').classList.remove('hidden');
        }

        let time = 5400;
        setInterval(() => {
            if(time <= 0) return;
            time--;
            let m = Math.floor(time/60); let s = time%60;
            document.getElementById('timer').innerText = `${m}:${s < 10 ? '0' : ''}${s}`;
        }, 1000);
    </script>
